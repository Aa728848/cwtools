## 测试 $A|B$ 缺省值语法的 scripted_effect

## 这个应该能被正确识别，参数名为 amount，缺省值为 10
test_effect_with_default = {
    test = {
        test_scripted_effect_params = {
            test_lhs = $lhs|OR$
            test_rhs = yes
        }
    }
}

## 多个参数都带缺省值
test_effect_multiple_defaults = {
    test = {
        test_scripted_effect_params = {
            test_lhs = $lhs|OR$
            test_rhs = $rhs|yes$
        }
    }
}

## 混合使用：有的有缺省值，有的没有
test_effect_mixed = {
    $target$ = {
        test_scripted_effect_params = {
            test_lhs = $lhs|OR$
            test_rhs = $rhs$
        }
    }
}

## 左侧使用缺省值语法
$effect_name|my_effect$ = {
    test = {
        test_scripted_effect_params = {
            test_lhs = OR
            test_rhs = yes
        }
    }
}
